PKG_NAME="libtorrent-rasterbar"
PKG_VERSION="0.16.10"
PKG_SITE="http://www.rasterbar.com/"
PKG_URL="https://www.oneevil.ru/sources/$PKG_NAME-$PKG_VERSION.tar.gz"
PKG_DEPENDS_TARGET="toolchain Python boost"

PKG_SECTION="torrenter"

PKG_AUTORECONF="no"

configure_target() {
  cd $PKG_BUILD

  export PYTHON_VERSION="2.7"
  export PYTHON_CPPFLAGS="-I$SYSROOT_PREFIX/usr/include/python$PYTHON_VERSION"
  export PYTHON_LDFLAGS="-L$SYSROOT_PREFIX/usr/lib/python$PYTHON_VERSION -lpython$PYTHON_VERSION"
  export PYTHON_SITE_PKG="$SYSROOT_PREFIX/usr/lib/python$PYTHON_VERSION/site-packages"

  ./configure --host=$TARGET_NAME \
              --build=$HOST_NAME \
              --with-openssl=$SYSROOT_PREFIX/usr \
              --with-boost-chrono=$SYSROOT_PREFIX/usr/lib \
              --with-boost-date_time=$SYSROOT_PREFIX/usr/lib \
              --with-boost-filesystem=$SYSROOT_PREFIX/usr/lib \
              --with-boost-graph=$SYSROOT_PREFIX/usr/lib \
              --with-boost-iostreams=$SYSROOT_PREFIX/usr/lib \
              --with-boost-locale=$SYSROOT_PREFIX/usr/lib \
              --with-boost-math_c99=$SYSROOT_PREFIX/usr/lib \
              --with-boost-math_c99f=$SYSROOT_PREFIX/usr/lib \
              --with-boost-math_c99l=$SYSROOT_PREFIX/usr/lib \
              --with-boost-math_tr1=$SYSROOT_PREFIX/usr/lib \
              --with-boost-math_tr1f=$SYSROOT_PREFIX/usr/lib \
              --with-boost-math_tr1l=$SYSROOT_PREFIX/usr/lib \
              --with-boost-prg_exec_monitor=$SYSROOT_PREFIX/usr/lib \
              --with-boost-program_options=$SYSROOT_PREFIX/usr/lib \
              --with-boost-python=$SYSROOT_PREFIX/usr/lib \
              --with-boost-random=$SYSROOT_PREFIX/usr/lib \
              --with-boost-regex=$SYSROOT_PREFIX/usr/lib \
              --with-boost-serialization=$SYSROOT_PREFIX/usr/lib \
              --with-boost-signals=$SYSROOT_PREFIX/usr/lib \
              --with-boost-system=$SYSROOT_PREFIX/usr/lib \
              --with-boost-thread=$SYSROOT_PREFIX/usr/lib \
              --with-boost-timer=$SYSROOT_PREFIX/usr/lib \
              --with-boost-unit_test_framework=$SYSROOT_PREFIX/usr/lib \
              --with-boost-wave=$SYSROOT_PREFIX/usr/lib \
              --with-boost-wserialization=$SYSROOT_PREFIX/usr/lib \
              --prefix=/usr \
              --disable-debug \
              --enable-python-binding
             
./make
}

post_makeinstall_target() {
  rm -f $INSTALL/usr/lib/libtorrent-rasterbar.a
  rm -f $INSTALL/usr/lib/libtorrent-rasterbar.la
  rm -f $INSTALL/usr/lib/pkgconfig/libtorrent-rasterbar.pc
  rm -fr $INSTALL/usr/include/libtorrent
}
